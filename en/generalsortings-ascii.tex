GeneralSortings(S, LEQ(_, _)) ==
  LET D == 1..Cardinality(S)
  IN  { seq \in [D -> S] : /\ S \subseteq {seq[i] : i \in D}
                           /\ \A i, j \in D : (i < j) => LEQ(seq[i], seq[j]) }
