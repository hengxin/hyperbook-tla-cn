Warshall(R) ==
  LET NR == NodesOf(R)
      RECURSIVE W(_)
      W(L) == IF L = {} 
                THEN R
                ELSE LET n  == CHOOSE node \in L : TRUE
                         WM == W(L \ {n})
                     IN  WM \cup {rs \in NR \X NR : 
                                   (<<rs[1], n>> \in WM) /\ (<<n, rs[2]>> \in WM)}
  IN  W(NR)
