RECURSIVE SortSet(_)
SortSet(S) == 
  IF S = {} THEN << >>
            ELSE LET s == CHOOSE ss \in S : \A t \in S : ss.key =< t.key
                 IN  <<s>> \o SortSet(S \ {s})
